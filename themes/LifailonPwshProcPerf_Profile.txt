function Set-EnvVar {
    $env:POSH_CPU_USE = $(Get-Process -id $($env:POSH_PID)).CPU
    $env:POSH_MEM_USE = $($(Get-Process -id $($env:POSH_PID)).WorkingSet / 1mb).ToString("0.00 Mb")
}
New-Alias -Name "Set-PoshContext" -Value "Set-EnvVar" -Scope Global -Force
